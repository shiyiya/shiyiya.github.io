## 前序

```ts
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     val: number
 *     left: TreeNode | null
 *     right: TreeNode | null
 *     constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.left = (left===undefined ? null : left)
 *         this.right = (right===undefined ? null : right)
 *     }
 * }
 */

function preorderTraversal(root: TreeNode | null): number[] {
  const res = []
  const stack = [root]

  if (root == null) return []
  while (stack.length) {
    const t1 = stack.pop()
    res.push(t1.val)

    if (t1.right != null) stack.push(t1.right)
    if (t1.left != null) stack.push(t1.left)
  }
  return res
}
```

## 中序

```ts
function inorderTraversal(root: TreeNode | null): number[] {
  const stack = []
  const res = []

  while (root != null || stack.length) {
    while (root != null) {
      stack.push(root)
      root = root.left
    }

    const t = stack.pop()
    res.push(t.val)
    root = t.right
  }
  return res
}
```

## 后序

```ts
function postorderTraversal(root: TreeNode | null): number[] {
  const res = []
  const stack = [root]

  if (root == null) return []
  while (stack.length) {
    const t1 = stack.pop()
    res.push(t1.val)

    if (t1.left != null) stack.push(t1.left)
    if (t1.right != null) stack.push(t1.right)
  }
  return res.reverse()
}
```

## 层序遍历

- https://leetcode-cn.com/leetbook/read/data-structure-binary-tree/xefh1i/

```ts
function levelOrder(root: TreeNode | null): number[][] {
  if (root == null) return []

  const queue = [root]
  const res = []

  while (queue.length > 0) {
    const size = queue.length
    const sub = []

    for (let i = 0; i < size; i++) {
      const top = queue.shift()
      sub.push(top.val)

      if (top.left != null) {
        queue.push(top.left)
      }
      if (top.right != null) {
        queue.push(top.right)
      }
    }
    res.push(sub)
  }
  return res
}
```

## 对称二叉树

```ts
// 递归
function isSymmetric(root: TreeNode | null): boolean {
  return check(root.left, root.right)
}

function check(left: TreeNode, right: TreeNode) {
  if (left == null && right == null) return true
  if (left == null || right == null || left.val != right.val) return false

  return check(left.left, right.right) && check(left.right, right.left)
}

// 队列
function isSymmetric(root: TreeNode | null): boolean {
  if (root == null) return true

  const queue = [root.left, root.right]

  while (queue.length) {
    const [left, right] = [queue.shift(), queue.shift()]
    if (left == null && right == null) continue
    if (left == null || right == null || left.val != right.val) return false

    queue.push(left.left, right.right, left.right, right.left)
  }
  return true
}
```
